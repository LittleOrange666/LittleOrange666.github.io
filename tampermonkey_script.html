// ==UserScript==
// @name         Codeforces mashup creater
// @namespace    http://tampermonkey.net/
// @version      0.1
// @description  try to take over the world!
// @author       You
// @match        https://codeforces.com/*
// @icon         https://codeforces.org/s/56111/favicon-96x96.png
// @grant        none
// ==/UserScript==

(function() {
    &#x27;use strict&#x27;;
    function inputevt(o,value){
        let evt = document.createEvent(&#x27;HTMLEvents&#x27;);
        evt.initEvent(&#x27;input&#x27;, true, true);
        o.value = value;
        o.dispatchEvent(evt);
    }
    function changeevt(o,value){
        let evt = document.createEvent(&#x27;HTMLEvents&#x27;);
        evt.initEvent(&#x27;change&#x27;, true, true);
        o.value = value;
        o.dispatchEvent(evt);
    }
    function evt(o,value){
        let evt = document.createEvent(&#x27;HTMLEvents&#x27;);
        evt.initEvent(value, true, true);
        o.dispatchEvent(evt);
    }
    function shuffle(array) {
        for (let i = array.length - 1; i &gt; 0; i--) {
            let j = Math.floor(Math.random() * (i + 1));
            [array[i], array[j]] = [array[j], array[i]];
        }
    }
    if(location.pathname.endsWith(&quot;problems/new&quot;)){
        let input = $(&quot;.problemTable .ac_input&quot;)[0];
        let area1 = $(&quot;&lt;input&gt;&quot;).addClass(&quot;ac_input&quot;);
        let area2 = $(&quot;&lt;input&gt;&quot;).addClass(&quot;ac_input&quot;);
        $(&quot;#pageContent&quot;).append(&quot;ranges:&quot;).append(area1).append(&quot;&lt;br&gt;&quot;);
        $(&quot;#pageContent&quot;).append(&quot;participants:&quot;).append(area2).append(&quot;&lt;br&gt;&quot;);
        let btn = $(&quot;&lt;button&gt;&quot;).text(&quot;run mashup creater&quot;);
        $(&quot;#pageContent&quot;).append(btn);
        btn.click(function(){
            let ranges = area1.val().split(&quot;,&quot;);
            let participants = area2.val().split(&quot;,&quot;);
            console.log(ranges, participants);
            localStorage.ranges = JSON.stringify(ranges);
            localStorage.participants = JSON.stringify(participants);
            localStorage.completed = JSON.stringify([]);
            localStorage.page_cnt = JSON.stringify({});
            window.open(&quot;https://codeforces.com/#mashup_run","_blank");
            var inter_id;
            inter_id = window.setInterval(function(){
                let completed = JSON.parse(localStorage.completed);
                if (completed.length == ranges.length){
                    window.clearInterval(inter_id);
                    shuffle(completed);
                    let idx = 0;
                    window.setInterval(function(){
                        if (idx==completed.length){
                            $(&quot;._MashupContestEditFrame_saveMashup .submit&quot;).click();
                        }else{
                            inputevt(input,completed[idx]);
                            idx++;
                            $(&quot;._MashupContestEditFrame_addProblem&quot;).submit();
                        }
                    },2000);
                }
            },1000);
        });
    }
    if (location.hash == &quot;#mashup_run&quot;){
        sessionStorage.mashup_run = 1;
        location.hash = &quot;&quot;;
        sessionStorage.state = &quot;prepare&quot;;
    }
    if (sessionStorage.mashup_run){
        function do_action(){
            window.setTimeout(function(){
                let ranges = JSON.parse(localStorage.ranges);
                let completed = JSON.parse(localStorage.completed);
                let participants = JSON.parse(localStorage.participants);
                let page_cnt = JSON.parse(localStorage.page_cnt);
                switch(sessionStorage.state){
                    case &quot;prepare&quot;:
                        if (completed.length&lt;ranges.length){
                            let cur = ranges[completed.length];
                            if (cur in page_cnt){
                                sessionStorage.state = &quot;choose&quot;;
                                location.href = &quot;/problemset/page/"+(Math.floor(page_cnt[cur]*Math.random())+1)+"?tags="+cur;
                            }else{
                                sessionStorage.state = &quot;count&quot;;
                                location.href = &quot;https://codeforces.com/problemset?tags="+cur;
                            }
                        }else{
                            window.close();
                        }
                        break;
                    case &quot;count&quot;:
                        page_cnt[location.search.substr(6)]=Math.max(1,+$(&quot;.pagination span:last&quot;).text()-1);
                        localStorage.page_cnt = JSON.stringify(page_cnt);
                        sessionStorage.state = &quot;prepare&quot;;
                        do_action();
                        break;
                    case &quot;choose&quot;:
                        {
                            let problems = $(&quot;.problems tr:not(.accepted-problem)&quot;);
                            let cur = ranges[completed.length];
                            if (problems.length==0) location.href = &quot;/problemset/page/"+(Math.floor(page_cnt[cur]*Math.random())+1)+"?tags="+cur;
                            else{
                                let got = problems[Math.floor(Math.random()*problems.length)].querySelector(&quot;.id&quot;).textContent.trim();
                                sessionStorage.got = got;
                                sessionStorage.state = &quot;verify&quot;;
                                sessionStorage.verify = &quot;0&quot;;
                                do_action();
                            }
                        }
                        break;
                    case &quot;verify&quot;:
                        {
                            let pos = sessionStorage.got.match(/[A-Z]/i).index;
                            let name = sessionStorage.got.substr(0,pos);
                            let pid = sessionStorage.got.substr(pos);
                            if (sessionStorage.verify==&quot;0&quot;){
                                let link = &quot;/contest/&quot;+name;
                                if (location.pathname != link){
                                    location.href = link;
                                }else{
                                    let faild = $(&quot;#pageContent sup&quot;).length&gt;0;
                                    $(&quot;.caption.titled&quot;).each(function(){
                                        if($(this).text().includes(&quot;Languages&quot;)){
                                            faild = true;
                                        }
                                    });
                                    if(faild){
                                        delete sessionStorage.got;
                                        delete sessionStorage.verify;
                                        sessionStorage.state = &quot;prepare&quot;;
                                        do_action();
                                    }else{
                                        sessionStorage.verify = &quot;1&quot;;
                                        sessionStorage.idx = &quot;0&quot;;
                                        do_action();
                                    }
                                }
                            }else if (sessionStorage.verify==&quot;1&quot;){
                                let idx = +sessionStorage.idx;
                                if (idx == participants.length){
                                    completed.push(sessionStorage.got);
                                    localStorage.completed = JSON.stringify(completed);
                                    delete sessionStorage.got;
                                    delete sessionStorage.verify;
                                    delete sessionStorage.idx;
                                    sessionStorage.state = &quot;prepare&quot;;
                                    do_action();
                                }else{
                                    let link = &quot;/submissions/&quot;+participants[idx];
                                    if (location.pathname != link){
                                        location.href = link;
                                    }else{
                                        if (!sessionStorage.choosecontest){
                                            changeevt(document.querySelector(&quot;#_ContestStatusFilterFormPart_chosenContest&quot;),&quot;specifyContest&quot;);
                                            inputevt($(&quot;#_ContestStatusFilterFormPart_contestIdAndNameField&quot;)[0],name);
                                            sessionStorage.choosecontest = &quot;1&quot;;
                                            $(&quot;form.status-filter input[type=&#x27;submit&#x27;]&quot;).click();
                                        }else{
                                            delete sessionStorage.choosecontest;
                                            let faild = false;
                                            $(&quot;.status-frame-datatable tr[data-submission-id] td[data-problemid]&quot;).each(function(){
                                                faild = true;
                                            })
                                            if(faild){
                                                delete sessionStorage.got;
                                                delete sessionStorage.verify;
                                                delete sessionStorage.idx;
                                                sessionStorage.state = &quot;prepare&quot;;
                                                do_action();
                                            }else{
                                                sessionStorage.idx = idx+1;
                                                do_action();
                                            }
                                        }
                                    }
                                }
                            }
                            break;
                        }
                }
            },2000);
        }
        do_action();
    }
})();
