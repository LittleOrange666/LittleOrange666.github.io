# 歷屆第1題
## 2016年3月
[成績指標](https://zerojudge.tw/ShowProblem?problemid=b964)
:::spoiler 提示
std::sort
:::
:::spoiler 題解
:::
:::spoiler C++範例代碼：
```cpp
#include <bits/stdc++.h>
using namespace std;
int main(){
    //輸入
    int n;
    cin >> n;
    vector<int> a(n);
    for (int i = 0; i < n;i++) cin >> a[i];

    //計算
    int big = -1000;
    int small = 1000;
    for (int i = 0; i < n;i++){
        if (a[i]>=60){
            small = min(small, a[i]);
        }else{
            big = max(big, a[i]);
        }
    }

    //排序
    sort(a.begin(), a.end());

    //輸出
    for (int i = 0; i < n;i++){
        cout << a[i];
        if (i<n-1) cout << " ";
    }
    cout << "\n";
    if (big==-1000)cout << "best case\n";
    else cout << big << "\n";
    if (small==1000)cout << "worst case\n";
    else cout << small << "\n";
}
```

## 2016年10月
[三角形辨別](https://zerojudge.tw/ShowProblem?problemid=c294)
:::spoiler 提示
std::sort
:::
:::spoiler 題解
:::
:::spoiler C++範例代碼：
```cpp
#include <bits/stdc++.h>
using namespace std;
int main(){
    //輸入
    vector<int> a(3);
    for (int i = 0; i < 3;i++){
        cin >> a[i];
    }

    //排序
    sort(a.begin(),a.end());

    //輸出
    for (int i = 0; i < 3;i++){
        cout << a[i];
        if (i<2) cout << " ";
    }
    cout << "\n";
    if (a[0]+a[1]<=a[2]){
        cout << "No\n";
    }else if (a[0]*a[0]+a[1]*a[1]<a[2]*a[2]){
        cout << "Obtuse\n";
    }else if (a[0]*a[0]+a[1]*a[1]==a[2]*a[2]){
        cout << "Right\n";
    }else if (a[0]*a[0]+a[1]*a[1]>a[2]*a[2]){
        cout << "Acute\n";
    }
}
```

## 2017年3月
[秘密差](https://zerojudge.tw/ShowProblem?problemid=c290)
:::spoiler 提示
std::string
:::
:::spoiler 題解
:::
:::spoiler C++範例代碼：
```cpp
#include <bits/stdc++.h>
using namespace std;
int main(){
    string num;
    cin >> num;
    int n = num.length();
    int a=0,b=0;
    for (int i = 1; i <= n; i++){
        if (i%2==1){
            a+=num[n-i]-'0';
        }else{
            b+=num[n-i]-'0';
        }
    }
    cout << abs(a-b) << "\n";
}
```
:::
:::spoiler Python範例代碼：

## 2017年10月
[邏輯運算子](https://zerojudge.tw/ShowProblem?problemid=c461)
:::spoiler 提示
布林運算
:::
:::spoiler 題解
:::
:::spoiler C++範例代碼：
```cpp
#include <bits/stdc++.h>
using namespace std;
int main(){
    int a,b,c;
    cin >> a >> b >> c;
    bool impossible = true;
    if (((a>0)&&(b>0))==c){
        impossible = false;
        cout << "AND\n";
    }
    if (((a>0)||(b>0))==c){
        impossible = false;
        cout << "OR\n";
    }
    if (((a>0)!=(b>0))==c){
        impossible = false;
        cout << "XOR\n";
    }
    if (impossible) cout << "IMPOSSIBLE\n";
}
```
:::
:::spoiler Python範例代碼：
```py
a,b,c = map(int,input().split())
impossible = True
if ((a>0) and (b>0))==c:
    impossible = False
    print("AND")
if ((a>0) or (b>0))==c:
    impossible = False
    print("OR")
if ((a>0) != (b>0))==c:
    impossible = False
    print("XOR")
if impossible:
    print("IMPOSSIBLE")
```

#include <bits/stdc++.h>

## 2020年1月
[猜拳](https://zerojudge.tw/ShowProblem?problemid=h026)
:::spoiler 提示
:::
:::spoiler 題解
:::
:::spoiler C++範例代碼：
```cpp
#include <bits/stdc++.h>
using namespace std;
int main(){
    vector<int> better(10,0);
    better[0] = 5;
    better[2] = 0;
    better[5] = 2;
    int f, n;
    cin >> f >> n;
    vector<int> y(n);
    for (int i = 0;i<n;i++){
        cin >> y[i];
    }
    for (int k = 1;k<=n;k++){
        cout << f << " ";
        if (f == better[y[k-1]]){
            cout << ": Won at round " << k << "\n";
            break;
        }
        if (y[k-1] == better[f]){
            cout << ": Lost at round " << k << "\n";
            break;
        }
        if (k>1&&y[k-1]==y[k-2]) f = better[y[k-1]];
        else f = y[k-1];
        if (k==n) cout << ": Drew at round " << k << "\n";
    }
}
```
:::
:::spoiler Python範例代碼：
```py
better = [0]*10
better[0] = 5
better[2] = 0
better[5] = 2
f = int(input())
n = int(input())
y = list(map(int,input().split()))
for k in range(1,n+1):
    print(f, end=" ")
    if f == better[y[k-1]]:
        print(f": Won at round {k}")
        break
    if y[k-1] == better[f]:
        print(f": Lost at round {k}")
        break
    if k>1 and y[k-1]==y[k-2]:
        f = better[y[k-1]]
    else:
        f = y[k-1]
    if k==n:
        print(f": Drew at round {k}")
```
